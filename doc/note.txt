	incr/decr/incrby/decrby

### key
	del key [key ...]
	key_next prefix [count]
	key_prev prefix [count]

	hgetall
	hlen
	hdel
	NOTSUPPORTED(hexists/hincrby/hincrbyfloat/hkeys/hsetnx/hvals/hscan)

	lindex
	lrange
	llen
